        // 选择选项

        function selectOption(index) {

            const q = questions[currentQuestion];

            const [left, right] = dimensionMap[q.dimension];

            

            if (index === 0) scores[left] += 2;

            else if (index === 1) scores[left] += 1;

            else if (index === 2) scores[right] += 1;

            else scores[right] += 2;

            

            currentQuestion++;

            

            if (currentQuestion < 16) {

                showQuestion();

            } else {

                showResult();

            }

        }


        // 计算结果

        function calculateResult() {

            const d1 = scores.S >= scores.A ? 'S' : 'A';

            const d2 = scores.C >= scores.E ? 'C' : 'E';

            const d3 = scores.U >= scores.F ? 'U' : 'F';

            const d4 = scores.I >= scores.M ? 'I' : 'M';

            

            let code = d1 + d2 + d3 + d4;

            

            if (!personalities[code]) {

                const balanced = [scores.S === scores.A, scores.C === scores.E, scores.U === scores.F, scores.I === scores.M];

                if (balanced.filter(b => b).length >= 2) {

                    code = 'SEUM';

                } else {

                    code = 'ACEFIM';

                }

            }

            

            return personalities[code];

        }
